-- ? http://stackoverflow.com/questions/10912337/how-to-show-comments-of-a-column-with-desc-operation

SELECT cc.table_name,
        tc.column_name,
        tc.nullable,
        tc.data_type
            || CASE
                    WHEN tc.data_type = 'NUMBER' AND tc.data_precision IS NOT NULL
                        THEN '(' || tc.data_precision || ',' || tc.data_scale || ')'
                    WHEN tc.data_type LIKE '%CHAR%'
                        THEN '(' || tc.data_length || ')'
                    ELSE NULL
            END TYPE,
        cc.comments
    FROM   user_col_comments cc
    JOIN   user_tab_columns  tc
        ON  cc.column_name = tc.column_name
            AND cc.table_name  = tc.table_name
    WHERE  cc.table_name
        IN (SELECT table_name FROM all_tables)

static class Log
{
    public static int error_level = 3;

    //  Error levels:
    //   4 Error
    //   3 Info
    //   2 Debug
    //   1 Trace

    public static void Error(string message)
    {
        if (error_level <= 4)
        {
            Console.WriteLine("Error: {0} (Error level: {1})", message, error_level);
        }
    }

    public static void Info(string message)
    {
        if (error_level <= 3)
        {
            Console.WriteLine("Info: {0} (Error level: {1})", message, error_level);
        }
    }

    public static void Debug(string message)
    {
        if (error_level <= 2)
        {
            Console.WriteLine("Debug: {0} (Error level: {1})", message, error_level);
        }
    }

    public static void Trace(string message)
    {
        if (error_level == 1)
        {
            Console.WriteLine("Trace: {0} (Error level: {1})", message, error_level);
        }
    }
}
